1. Create topic: 
  bin/kafka-topics.sh --create --bootstrap-server localhost:9094 --topic ankit-demo --partitions 3 --replication-factor 3

2. List existing topics:
  bin/kafka-topics.sh --list --bootstrap-server localhost:9093
  
3. Describe a topics:
  bin/kafka-topics.sh --bootstrap-server localhost:9093 --describe --topic ankit-demo
  
4. Produce message:
  bin/kafka-console-producer.sh --bootstrap-server localhost:9094 --topic ankit-demo
  
5. Consume message:
  bin/kafka-console-consumer.sh --bootstrap-server localhost:9094 --topic ankit-demo
